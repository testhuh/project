### 1️⃣ **멜론 크롤링 → CSV 생성**

```python
# melon_crawler.py

from playwright.sync_api import sync_playwright
import pandas as pd

class MelonCrawler:
    def collect_songs(self):
        songs = []
        
        with sync_playwright() as p:
            browser = p.chromium.launch(headless=False)
            page = browser.new_page()
            
            # 멜론 공유 링크 접속 (로그인 불필요)
            page.goto(os.getenv('MELON_PLAYLIST_URL'))
            page.wait_for_load_state('networkidle')
            
            # 곡 목록 파싱
            song_elements = page.query_selector_all('tr.lst50')
            
            for element in song_elements:
                title = element.query_selector('span.ellipsis').text_content()
                artist = element.query_selector_all('span.ellipsis')[1].text_content()
                
                songs.append({
                    'title': title.strip(),
                    'artist': artist.strip(),
                    'search_query': f"{title} {artist}"
                })
            
            browser.close()
        
        # CSV 저장
        df = pd.DataFrame(songs)
        df.to_csv('melon_disney_ost_complete_list.csv', index=False)
        return songs

# 결과 CSV:
# title,artist,search_query
# Let It Go,Idina Menzel,Let It Go Idina Menzel
# Into the Unknown,Idina Menzel,Into the Unknown Idina Menzel
# ...
```

### 2️⃣ **CSV → YouTube 검색 → Video ID 추출**

```python
# youtube_search.py

class YouTubeSearcher:
    def search_and_save(self, csv_file):
        df = pd.read_csv(csv_file)
        results = []
        
        with sync_playwright() as p:
            browser = p.chromium.launch(headless=False)
            page = browser.new_page()
            
            for idx, row in df.iterrows():
                search_query = row['search_query']
                
                # YouTube 검색
                search_url = f"https://www.youtube.com/results?search_query={search_query}"
                page.goto(search_url)
                page.wait_for_load_state('networkidle')
                
                # 첫 번째 영상 클릭
                try:
                    first_video = page.query_selector('a#video-title-link')
                    if first_video:
                        video_url = first_video.get_attribute('href')
                        video_id = video_url.split('v=')[1].split('&')[0]
                        
                        results.append({
                            'title': row['title'],
                            'artist': row['artist'],
                            'video_id': video_id,
                            'youtube_url': f'https://www.youtube.com/watch?v={video_id}',
                            'status': '✅'
                        })
                        
                        # GUI 업데이트
                        update_gui(f"✅ [{idx+1}/18] {row['title']} 검색됨")
                except Exception as e:
                    results.append({
                        'title': row['title'],
                        'artist': row['artist'],
                        'video_id': 'N/A',
                        'youtube_url': 'N/A',
                        'status': '❌'
                    })
                    update_gui(f"❌ [{idx+1}/18] {row['title']} 검색 실패: {str(e)}")
            
            browser.close()
        
        # CSV 저장
        df_results = pd.DataFrame(results)
        df_results.to_csv('youtube_saveable_list.csv', index=False)
        return results

# 결과 CSV:

# Into the Unknown | Idina Menzel | PxYEm1rJdCo | https://www.youtube.com/watch?v=PxYEm1rJdCo | ✅
# ...
```

### 3️⃣ **YouTube 재생목록 생성**

```python
# youtube_playlist.py

class YouTubePlaylistManager:
    def create_playlist(self):
        with sync_playwright() as p:
            browser = p.chromium.launch(headless=False)
            page = browser.new_page()
            
            # YouTube 로그인
            page.goto('https://www.youtube.com')
            # ... 자동 로그인 (email, password는 .env에서)
            
            # 재생목록 생성
            page.goto('https://www.youtube.com/me/playlists')
            page.click('button:has-text("+ 새 재생목록 만들기")')
            
            # 이름 입력
            page.fill(
                'input[aria-label="재생목록 제목"]',
                f"Disney OST Playlist - {date.today()}"
            )
            
            # 공개 설정
            page.click('button:has-text("공개")')
            page.click('button:has-text("만들기")')
            
            page.wait_for_load_state('networkidle')
            
            # 재생목록 URL 추출
            playlist_url = page.url
            # 예: https://www.youtube.com/playlist?list=PLxxxxxxxxxx
            
            browser.close()
            
            return playlist_url
```

### 4️⃣ **CSV의 Video ID로 재생목록에 곡 추가 (GUI 실시간 모니터링)**

```python
# youtube_add_songs.py

class YouTubePlaylistUpdater:
    def add_songs_from_csv(self, csv_file, playlist_id):
        df = pd.read_csv(csv_file)
        
        with sync_playwright() as p:
            browser = p.chromium.launch(headless=False)
            page = browser.new_page()
            
            success_count = 0
            fail_count = 0
            
            for idx, row in df.iterrows():
                if row['status'] != '✅' or row['video_id'] == 'N/A':
                    fail_count += 1
                    # GUI 업데이트: 건너뜀
                    continue
                
                try:
                    video_id = row['video_id']
                    
                    # 재생목록에 직접 추가 (Playwright로 클릭)
                    page.goto(f'https://www.youtube.com/watch?v={video_id}')
                    page.wait_for_load_state('networkidle')
                    
                    # 저장 버튼 클릭
                    page.click('button[aria-label="저장"]')
                    page.wait_for_timeout(500)
                    
                    # 재생목록에 추가
                    page.click('text=재생목록에 추가')
                    page.wait_for_timeout(500)
                    
                    # 대상 재생목록 선택
                    page.click(f'text=Disney OST Playlist')
                    page.wait_for_timeout(1000)
                    
                    success_count += 1
                    
                    # GUI 실시간 업데이트
                    progress = int((idx + 1) / len(df) * 100)
                    update_gui(f"✅ [{idx+1}/{len(df)}] {row['title']} 추가됨")
                    update_progress(progress)
                    
                except Exception as e:
                    fail_count += 1
                    update_gui(f"❌ [{idx+1}/{len(df)}] {row['title']} 추가 실패: {str(e)}")
            
            browser.close()
            
            return {
                'success': success_count,
                'fail': fail_count,
                'total': len(df),
                'success_rate': f"{success_count/len(df)*100:.0f}%"
            }
```

## 💻 메인 코드 구조

```python
# main.py

import os
from dotenv import load_dotenv
from melon_crawler import MelonCrawler
from youtube_search import YouTubeSearcher
from youtube_playlist import YouTubePlaylistManager
from youtube_add_songs import YouTubePlaylistUpdater
from gui import YouTubeAutomationGUI
import sys
from PyQt5.QtWidgets import QApplication

load_dotenv()

class AutomationSystem:
    def __init__(self):
        self.gui = YouTubeAutomationGUI()
    
    def run(self):
        self.gui.start_button.clicked.connect(self.start_automation)
        self.gui.show()
        return sys.exit(app.exec_())
    
    def start_automation(self):
        try:
            # Step 1: 멜론 크롤링 → CSV 생성
            self.gui.log_emit("🎵 [Step 1] 멜론 크롤링 중...")
            crawler = MelonCrawler(
                url=os.getenv('MELON_PLAYLIST_URL'),
                gui_callback=self.gui.log_emit
            )
            songs = crawler.collect_songs()
            self.gui.log_emit(f"✅ [Step 1] {len(songs)}곡 수집 완료")
            self.gui.log_emit("   └─ CSV 저장: melon_disney_ost_complete_list.csv")
            
            # Step 2: CSV → YouTube 검색 → Video ID 추출
            self.gui.log_emit("🔍 [Step 2] YouTube 검색 중...")
            searcher = YouTubeSearcher(
                gui_callback=self.gui.log_emit,
                progress_callback=self.gui.update_progress
            )
            results = searcher.search_and_save('melon_disney_ost_complete_list.csv')
            success_count = len([r for r in results if r['status'] == '✅'])
            self.gui.log_emit(f"✅ [Step 2] {success_count}곡 검색 완료")
            self.gui.log_emit("   └─ CSV 저장: youtube_saveable_list.csv")
            
            # Step 3: 재생목록 생성
            self.gui.log_emit("📺 [Step 3] 재생목록 생성 중...")
            manager = YouTubePlaylistManager(
                email=os.getenv('YOUTUBE_EMAIL'),
                password=os.getenv('YOUTUBE_PASSWORD'),
                gui_callback=self.gui.log_emit
            )
            playlist_url = manager.create_playlist()
            self.gui.log_emit(f"✅ [Step 3] 재생목록 생성 완료")
            self.gui.log_emit(f"   └─ {playlist_url}")
            
            # Step 4: CSV의 Video ID로 재생목록에 곡 추가
            self.gui.log_emit("🎵 [Step 4] 곡 추가 중... (실시간 모니터링)")
            updater = YouTubePlaylistUpdater(
                gui_callback=self.gui.log_emit,
                progress_callback=self.gui.update_progress
            )
            result = updater.add_songs_from_csv(
                'youtube_saveable_list.csv',
                playlist_url
            )
            
            # 최종 결과
            self.gui.log_emit(f"🎉 [완료] {result['success']}/{result['total']} 곡 추가됨 ({result['success_rate']})")
            self.gui.log_emit(f"🔗 재생목록: {playlist_url}")
            
        except Exception as e:
            self.gui.log_emit(f"❌ 오류: {str(e)}")

if __name__ == '__main__':
    app = QApplication(sys.argv)
    system = AutomationSystem()
    system.run()
```
